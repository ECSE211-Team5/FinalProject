192  public static void searchArea(Navigation navigation, RingSearcher searcher, ColorCalibrator.Color targetColor) {
193  int counter = 0;
194  for (double i = LLx + 0.7; i < URx + 1; ) {
195  // if we are at the first, third...etc verticle zone, search up
196  if (counter % 2 == 0) {
197  for (double j = LLy + 0.7; j < URy; j++) {
198  navigation.travelTo(i, j, true);
199  navigation.turnTo(0, false);
200  if (searcher.search(55, targetColor)) return;
201  navigation.turnTo(0, false);
202  if (searcher.search(-55, targetColor)) return;
203  navigation.turnTo(0, false);
204  }
205  } else {
206  //when we are at the second, fourth... verticle zone, search down
207  for (double j = URy - 1 + 0.7; j > LLy; j--) {
208  navigation.travelTo(i, j, true);
209  navigation.turnTo(180, false);

211  navigation.turnTo(180, false);
212  if (searcher.search(225, targetColor)) return;
213  navigation.turnTo(180, false);
214  }
215  }
216  counter ++;
217  i = (i+2 < URx+1)? i+2 : (URx-0.7);
218  }
